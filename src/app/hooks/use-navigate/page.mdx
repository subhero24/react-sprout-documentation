# useNavigate

A hook to navigate programmatically.

```javascript
import { useNavigate } from 'react-sprout';

let [navigate, busy, loading, navigations] = useNavigate();
```

<Properties>
	<Property name="navigate" type="function">
		A function to navigate programmatically.
	</Property>
	<Property name="busy" type="boolean">
		A boolean indicating whether navigations are currently busy.
	</Property>
	<Property name="loading" type="boolean">
		A boolean indicating whether navigations are currently loading.
	</Property>
	<Property name="navigations" type="Array">
		An array of busy navigations.
	</Property>
</Properties>

### Navigate

```javascript
navigate(options);
navigate(to, options);
```

<Properties>
	<Property name="to" type="string">
		The target url of the navigation
	</Property>
	<Property name="options" type="object">
		The navigation options
	</Property>
</Properties>

### Navigate options

<Properties>
	<Property name="options.relative" type="boolean" defaultValue="true">
		A boolean indicating whether the target url is relative to the route.
	</Property>
	<Property name="options.method" type="string" defaultValue="GET">
		A string indicating the type of the navigation.
	</Property>
	<Property name="options.data" type="any">
		The data associated with the navigation
	</Property>
	<Property name="options.push" type="boolean">
		A boolean indicating whether a new entry in the history stack should be
		pushed.
	</Property>
	<Property name="options.replace" type="boolean">
		A boolean indicating whether the current entry in the history stack should
		be replaced.
	</Property>
	<Property name="options.title" type="string">
		The document title after navigation.
	</Property>
	<Property name="options.state" type="any">
		The history state after navigation.
	</Property>
	<Property name="options.sticky" type="boolean" defaultValue="false">
		A boolean indicating whether the navigation is sticky.
	</Property>
	<Property name="options.reload" type="boolean" defaultValue="false">
		A boolean indicating whether to reload all data for the navigation.
	</Property>
	<Property name="options.cache" type="boolean" defaultValue="false">
		A boolean indicating whether the current page should be kept in the
		back/forward cache.
	</Property>
	<Property name="options.onNavigate" type="function">
		A callback that is invoked when a navigation is initiated.
	</Property>
	<Property name="options.onCanceled" type="function">
		A callback that is invoked when a navigation is canceled.
	</Property>
	<Property name="options.onNavigateStart" type="function">
		A callback that is invoked when a navigation is not canceled.
	</Property>
	<Property name="options.onActionError" type="function">
		A callback that is invoked when a navigation action throws an error.
	</Property>
	<Property name="options.onAborted" type="function">
		A callback that is invoked when a navigation is aborted.
	</Property>
	<Property name="options.onNavigateEnd" type="function">
		A callback that is invoked when a navigation is finished.
	</Property>
</Properties>
