# Router

The Router component is created by calling the default export with a routes configuration.

```javascript
import Routes from 'react-sprout';

let Router = Routes(
	<App>
		<Todos path="todos" />
		<Todo path="todo/:todoId" />
	</App>,
);
```

The resulting Router component can be rendered anywhere in your application.

```javascript
function Application() {
	return <Router />;
}
```

### Properties

<Properties>
	<Property name="request" type="Request">
		A request when using the router as a controlled component, making the router
		independent of the browser.
	</Property>
	<Property name="defaultRequest" type="Request">
		An initial request when using the router as an uncontrolled component,
		making the router independent of the browser.
	</Property>
	<Property name="sticky" type="boolean" defaultValue="false">
		The default sticky behavior for all navigations
	</Property>
	<Property name="dataTransform" type="function">
		A function to automatically transform navigation data.
	</Property>
	<Property
		id="delayLoadingMs"
		name="delayLoadingMs"
		type="integer"
		defaultValue="30"
	>
		The amount of milliseconds to delay a loading indicator.
	</Property>
	<Property
		id="minimumLoadingMs"
		name="minimumLoadingMs"
		type="integer"
		defaultValue="500"
	>
		The minimum amount of milliseconds a loading indicator is visible.
	</Property>
	<Property name="defaultFormMethod" type="string" defaultValue="GET">
		The default method property for forms.
	</Property>
	<Property name="onNavigate" type="function">
		A callback that is invoked when a navigation is initiated.
	</Property>
	<Property name="onCancel" type="function">
		A callback that is invoked when a navigation is canceled.
	</Property>
	<Property name="onNavigateStart" type="function">
		A callback that is invoked when a navigation is not canceled.
	</Property>
	<Property name="onActionError" type="function">
		A callback that is invoked when a navigation action throws an error.
	</Property>
	<Property name="onAbort" type="function">
		A callback that is invoked when a navigation is aborted.
	</Property>
	<Property name="onNavigateEnd" type="function">
		A callback that is invoked when a navigation is finished.
	</Property>
</Properties>
