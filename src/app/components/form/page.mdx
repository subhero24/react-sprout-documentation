# Form

The Form component is used to navigate around with user input. It has all the same properties as the native [form](https://developer.mozilla.org/en-US/docs/Web/HTML/Element/form) element, and a few additional ones.

```javascript
import { Form } from 'react-sprout';
```

### Additional properties

<Properties>
	<Property name="relative" type="boolean" defaultValue="true">
		A boolean indicating whether the action url is relative to the route.
	</Property>
	<Property name="push" type="boolean">
		A boolean indicating whether a new entry in the history stack should be
		pushed.
	</Property>
	<Property name="replace" type="boolean">
		A boolean indicating whether the current entry in the history stack should
		be replaced.
	</Property>
	<Property name="title" type="string">
		The title for the new page.
	</Property>
	<Property name="state" type="any">
		The history state after navigation.
	</Property>
	<Property name="sticky" type="boolean" defaultValue="false">
		A boolean indicating whether the navigation is sticky.
	</Property>
	<Property name="reload" type="boolean" defaultValue="false">
		A boolean indicating whether to reload all data for the navigation.
	</Property>
	<Property name="cache" type="boolean" defaultValue="false">
		A boolean indicating whether the current page should be kept in the
		back/forward cache.
	</Property>
	<Property name="onNavigate" type="function">
		A callback that is invoked when a navigation is initiated.
	</Property>
	<Property name="onCanceled" type="function">
		A callback that is invoked when a navigation is canceled.
	</Property>
	<Property name="onNavigateStart" type="function">
		A callback that is invoked when a navigation is not canceled.
	</Property>
	<Property name="onActionError" type="function">
		A callback that is invoked when a navigation action throws an error.
	</Property>
	<Property name="onAborted" type="function">
		A callback that is invoked when a navigation is aborted.
	</Property>
	<Property name="onNavigateEnd" type="function">
		A callback that is invoked when a navigation is finished.
	</Property>
</Properties>
